// Binary Search - Iterative Method #include <stdio.h> // Binary search function (iterative) int binarySearch(int arr[], int n, int key) { int left = 0; // ‡∑Ä‡∂∏‡∑ä ‡∂¥‡∑É‡∑í‡∂±‡∑ä ‡∂Ü‡∂ª‡∂∏‡∑ä‡∂∑‡∂ö index int right = n - 1; // ‡∂Ø‡∂ö‡∑î‡∂´‡∑î ‡∂¥‡∑É‡∑í‡∂±‡∑ä ‡∂Ö‡∑Ä‡∑É‡∑è‡∂± index while (left <= right) { // ‡∑Ä‡∂∏‡∑ä ‡∑É‡∑Ñ ‡∂Ø‡∂ö‡∑î‡∂´‡∑î cross ‡∂±‡∑ú‡∑Ä‡∂± ‡∂≠‡∑î‡∂ª‡∑î int mid = left + (right - left) / 2; // ‡∂∏‡∑ê‡∂Ø index ‡∂ë‡∂ö ‡∂ú‡∂´‡∂±‡∂∫ ‡∂ö‡∂ª‡∂±‡∑ä‡∂± // ‡∂∏‡∑ê‡∂Ø element ‡∂ë‡∂ö key ‡∂ë‡∂ö ‡∑É‡∂∏‡∂ü ‡∂¥‡∂ª‡∑ì‡∂ö‡∑ä‡∑Ç‡∑è ‡∂ö‡∂ª‡∂±‡∑ä‡∂± if (arr[mid] == key) { // ‡∑Ñ‡∂∏‡∑î ‡∑Ä‡∑î‡∂´‡∑è ‡∂±‡∂∏‡∑ä return mid; // index ‡∂ë‡∂ö return ‡∂ö‡∂ª‡∂±‡∑ä‡∂± } // key ‡∂ë‡∂ö ‡∑Ä‡∑ê‡∂©‡∑í ‡∂±‡∂∏‡∑ä ‡∑Ä‡∂∏‡∑ä ‡∂∑‡∑è‡∂ú‡∂∫ ignore ‡∂ö‡∂ª‡∂±‡∑ä‡∂± if (arr[mid] < key) { // ‡∂∏‡∑ê‡∂Ø ‡∑Ä‡∂ß‡∑í‡∂±‡∑è‡∂ö‡∂∏ key ‡∂ë‡∂ö‡∂ß ‡∑Ä‡∂©‡∑è ‡∂Ö‡∂©‡∑î ‡∂±‡∂∏‡∑ä left = mid + 1; // ‡∂Ø‡∂ö‡∑î‡∂´‡∑î ‡∂∑‡∑è‡∂ú‡∂∫‡∑ö search ‡∂ö‡∂ª‡∂±‡∑ä‡∂± } // key ‡∂ë‡∂ö ‡∂Ö‡∂©‡∑î ‡∂±‡∂∏‡∑ä ‡∂Ø‡∂ö‡∑î‡∂´‡∑î ‡∂∑‡∑è‡∂ú‡∂∫ ignore ‡∂ö‡∂ª‡∂±‡∑ä‡∂± else { // ‡∂∏‡∑ê‡∂Ø ‡∑Ä‡∂ß‡∑í‡∂±‡∑è‡∂ö‡∂∏ key ‡∂ë‡∂ö‡∂ß ‡∑Ä‡∂©‡∑è ‡∑Ä‡∑ê‡∂©‡∑í ‡∂±‡∂∏‡∑ä right = mid - 1; // ‡∑Ä‡∂∏‡∑ä ‡∂∑‡∑è‡∂ú‡∂∫‡∑ö search ‡∂ö‡∂ª‡∂±‡∑ä‡∂± } } return -1; // ‡∑Ñ‡∂∏‡∑î ‡∑Ä‡∑ô‡∂Ω‡∑è ‡∂±‡∑ê‡∂≠‡∑ä‡∂±‡∂∏‡∑ä -1 return } // Main function int main() { int arr[] = {11, 12, 22, 25, 34, 64, 90}; // sorted array ‡∂ë‡∂ö‡∂ö‡∑ä int n = sizeof(arr) / sizeof(arr[0]); // array size int key = 25; // ‡∑É‡∑ô‡∑Ä‡∑í‡∂∫ ‡∂∫‡∑î‡∂≠‡∑î ‡∂Ö‡∂ú‡∂∫ printf("Sorted Array: "); for (int i = 0; i < n; i++) { printf("%d ", arr[i]); } printf("\n"); int result = binarySearch(arr, n, key); // binary search function call if (result == -1) { printf("Element %d not found\n", key); } else { printf("Element %d found at index %d\n", key, result); } return 0; }

Binary Search - Recursive Implementation

// Binary Search - Recursive Method #include <stdio.h> // Binary search function (recursive) int binarySearchRecursive(int arr[], int left, int right, int key) { if (right >= left) { // valid range ‡∂ë‡∂ö‡∂ö‡∑ä ‡∂≠‡∑í‡∂∂‡∑ö ‡∂±‡∂∏‡∑ä int mid = left + (right - left) / 2; // ‡∂∏‡∑ê‡∂Ø index ‡∂ú‡∂´‡∂±‡∂∫ ‡∂ö‡∂ª‡∂±‡∑ä‡∂± // ‡∂∏‡∑ê‡∂Ø element ‡∂ë‡∂ö‡∂∏ key ‡∂ë‡∂ö ‡∂±‡∂∏‡∑ä if (arr[mid] == key) { return mid; // index ‡∂ë‡∂ö return ‡∂ö‡∂ª‡∂±‡∑ä‡∂± } // ‡∂∏‡∑ê‡∂Ø element ‡∂ë‡∂ö key ‡∂ë‡∂ö‡∂ß ‡∑Ä‡∂©‡∑è ‡∑Ä‡∑í‡∑Å‡∑è‡∂Ω ‡∂±‡∂∏‡∑ä ‡∑Ä‡∂∏‡∑ä ‡∂∑‡∑è‡∂ú‡∂∫‡∑ö search if (arr[mid] > key) { return binarySearchRecursive(arr, left, mid - 1, key); } // ‡∂∏‡∑ê‡∂Ø element ‡∂ë‡∂ö key ‡∂ë‡∂ö‡∂ß ‡∑Ä‡∂©‡∑è ‡∂ö‡∑î‡∂©‡∑è ‡∂±‡∂∏‡∑ä ‡∂Ø‡∂ö‡∑î‡∂´‡∑î ‡∂∑‡∑è‡∂ú‡∂∫‡∑ö search return binarySearchRecursive(arr, mid + 1, right, key); } return -1; // ‡∑Ñ‡∂∏‡∑î ‡∂±‡∑ú‡∑Ä‡∑î‡∂´‡∑è ‡∂±‡∂∏‡∑ä -1 } // Main function int main() { int arr[] = {2, 3, 4, 10, 40, 50, 60}; // sorted array ‡∂ë‡∂ö‡∂ö‡∑ä int n = sizeof(arr) / sizeof(arr[0]); // array size int key = 10; // ‡∑É‡∑ô‡∑Ä‡∑í‡∂∫ ‡∂∫‡∑î‡∂≠‡∑î ‡∂Ö‡∂ú‡∂∫ int result = binarySearchRecursive(arr, 0, n - 1, key); if (result == -1) { printf("Element not found\n"); } else { printf("Element found at index %d\n", result); } return 0; }
